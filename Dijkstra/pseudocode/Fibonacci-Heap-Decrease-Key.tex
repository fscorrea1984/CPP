\begin{algorithm}[h!]
  \DontPrintSemicolon
  \SetKwFunction{FFibonacciHeapDecreaseKey}{FibonacciHeapDecreaseKey}
  \SetKwProg{Fn}{Function}{}{}

  \Fn{\FFibonacciHeapDecreaseKey{$H,x,k$}}{
    \If{$k > key[x]$} {
      error ``new key is greater than the current key''\;
    }
    $key[x] := k$\;
    $y := p[x]$\;
    \If{$y <> NIL$ and $key[x] < key[y]$} {
      $Cut(H,x,y)$\;
      $Cascading-Cut(H,y)$\;
    }
    \If{$key[x] < key[min[H]]$} {
      $min[H] := x$\;
    }
  }
  
  \smallskip
\end{algorithm}
