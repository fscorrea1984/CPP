\begin{algorithm}[h!]
  \DontPrintSemicolon
  \SetKwFunction{FConsolidate}{Consolidate}
  \SetKwProg{Fn}{Function}{}{}

  \Fn{\FConsolidate{$H$}}{
    \For{$i : 0$ \KwTo $D(n[H])$} {
      $A[i] := NIL$\;
    }
    \ForEach{node $x$ in the root list of $H$} {
      $x := w$\;
      $d := degree[x]$\;
      \While{$A[i] <> NIL$} {
        $y := A[d]$\;
        \If{$key[x] > key[y]$} {
          $exchange x \leftrightarrow y $\;
        }
        $Fibonacci-Heap-Link(H,y,x)$\;
        $A[d] := NIL$\;
        $d := d + 1$\;
      }
      $A[d] := x$\;
    }
    $min[H] := NIL$\;
    \For{$i := 0$ \KwTo $D(n[H])$} {
      \If{$A[i] <> NIL$} {
        add $A[i]$ to the root list of $H$\;
        \lIf{$min[H] = NIL$ or $key[A[i]] < key[min[H]]$} {
          $min[H] := A[i]$\;
        }
      }
    }
  }
  
  \smallskip
\end{algorithm}
