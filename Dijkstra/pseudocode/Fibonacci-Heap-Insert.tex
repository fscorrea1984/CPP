\begin{algorithm}[h!]
  \DontPrintSemicolon
  \SetKwFunction{FFibonacciHeapInsert}{FibonacciHeapInsert}
  \SetKwProg{Fn}{Function}{}{}

  \Fn{\FFibonacciHeapInsert{$H,x$}}{
    $degree[x] := 0$\;
    $p[x] := NIL$\;
    $child[x] := NIL$\;
    $left[x] := x$\;
    $right[x] := x$\;
    $mark[x] := FALSE$\;
    concatenate the root list containing $x$ with root list $H$\;
    \If{$min[H] = NIL$ or $key[x] < key[min[H]]$} {
      $min[H] := x$\;
    }
    $n[H] = n[H] + 1$
  }
  
  \smallskip
\end{algorithm}
