\begin{algorithm}[h!]
  \DontPrintSemicolon
  \SetKwFunction{FFibonacciHeapLink}{FibonacciHeapLink}
  \SetKwProg{Fn}{Function}{}{}

  \Fn{\FFibonacciHeapLink{$H,y,x$}}{
    remove $y$ from the root list of $H$\;
    make $y$ a child of $x$\;
    $degree[x] := degree[x] + 1$\;
    $mark[y] := FALSE$\;
  }
  
  \smallskip
\end{algorithm}
