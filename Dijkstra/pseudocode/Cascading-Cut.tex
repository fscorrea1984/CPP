\begin{algorithm}[h!]
  \DontPrintSemicolon
  \SetKwFunction{FCascadingCut}{CascadingCut}
  \SetKwProg{Fn}{Function}{}{}

  \Fn{\FCascadingCut{$H,y$}}{
    $z := p[y]$\;
    \If{$z <> NIL$} {
      \lIf{$mark[y] = FALSE$} {
        $mark[y] := TRUE$\;
      }
      \lElse{
        $Cut(H,y,z)$\;
        $Cascading-Cut(H,z)$\;
      }
    }
  }
  
  \smallskip
\end{algorithm}
